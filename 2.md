###7) revisiting demo

```python
name = "mbox-short.txt"
handle = open(name, 'r')
text = handle.read()
words = text.split() # splits into a list

counts = dict()
for word in words:
    counts[word] = counts.get(word,0) + 1 # assign key / value(counts) into the dictionary

bigcount = None
bigword = None
for word, count in counts.items(): # iterating through the dictionary 
    if bigcount is None or count > bigcount:
        bigword = word
        bigcount = count
print bigword, bigcount
```

###8) tuples simple examples

```python
x = ('Glenn', 'Sally', 'Joseph') # tuples are like lists
print "print x[2]:", x[2]
y = (1,9,2)
print "y:", y
print "max:", max(y)
for iter in y:
    print iter
# major difference is that tuples are immutable as opposed to list's mutability

# cannot sort, append, reverse (as tuples are immutable)

# tuples are more efficient; preferred over lists when using "temporary" variables

# more tuple assignment
(x,y) = (4, 'fred')
print x,y
a,b = 99,98
print a,b

# tuples and dictionaries
d = dict()
d['csev'] = 2
d['cwen'] = 4
for k,v in d.items():
    print k,v
tups = d.items()
print tups # list of tuples

print (0,1,2) < (5,1,2) # tuples are comparable (each element at a time)

# we can sort LISTS of tuples
d = {'a':10, 'b':1, 'c':22}
t = d.items()
print "list of tuples:",t
t.sort()
print "sorted list of tuples:",t

u = sorted(d.items()) # sorted takes sequence and returns sorted sequence

for k,v in sorted(d.items()): # sorts by key not value
    print k,v

# sorting by value
tmp = list()
for k,v in d.items():
    tmp.append((v,k)) # reverses key and value so that old key becomes new value, and vice-versa
tmp.sort(reverse=True)
print tmp
```
